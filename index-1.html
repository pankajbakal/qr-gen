Usage Example 
// Include the library
// <script src="logopay.js"></script>

// ── QR Code with logo ──────────────────────
const img = new Image();
img.src = 'logo.png';
img.onload = () => {
  const qr = LogoPay.QRCode({
    text:   'PMT|USD|0000010000|MCH-12345',
    image:  img,
    size:   400,         // pixels
    darkColor: '#000000',
    cornerRadius: 0.15,
  });
  document.body.appendChild(qr.canvas);
  qr.download('payment-qr', 'png');  // or 'svg'
  qr.download('payment-qr', 'svg');
};

// ── Barcode ────────────────────────────────
const bc = LogoPay.Barcode({
  text:   'INV-2024-0088',
  image:  img,
  width:  600,
  height: 150,
  barScale: 3,
});
bc.download('barcode', 'svg');

// ── ImageTool — resize + convert ───────────
