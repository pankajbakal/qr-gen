LogoPay.js

Logo-infused QR Code & Barcode Generation — Example Application

QR Code LogoPay.QRCode

// 1. Include the library
// <script src="logopay.js"></script>

const img = new Image();
img.src = 'logo.png';
img.onload = () => {
  const qr = LogoPay.QRCode({
    text:         'https://pay.example.com/invoice/0088',
    image:        img,          // logo fills light modules
    size:         400,         // output px (square)
    darkColor:    '#000000',
    cornerRadius: 0.15,        // rounded module corners
  });

  document.body.appendChild(qr.canvas);   // display
  qr.download('qr-code', 'png');          // save PNG
  qr.download('qr-code', 'svg');          // save SVG

  // Meta info: version, size, matrix
  console.log(qr.meta);
};

Code-128 Barcode LogoPay.Barcode

const bc = LogoPay.Barcode({
  text:      'INV-2024-0088',
  image:     img,          // optional logo
  width:     600,
  height:    150,
  barScale:  3,           // px per narrow bar unit
  darkColor: '#000000',
  showText:  true,         // render text below bars
});

document.body.appendChild(bc.canvas);
bc.download('barcode', 'svg');

Image Tool LogoPay.ImageTool

// Resize & convert a File from <input type="file">
const tool = LogoPay.ImageTool();
tool.load(fileInput.files[0])
  .then(t => t.resize(512, 512, 'fill'))
  .then(t => {
    t.download('thumbnail', 'png');
    t.download('thumbnail', 'svg');   // SVG with embedded raster
    t.download('thumbnail', 'webp');
  });
